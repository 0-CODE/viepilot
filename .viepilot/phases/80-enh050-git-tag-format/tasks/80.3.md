### 1. tests/unit/vp-enh050-git-tag-format.test.js

```javascript
const fs = require('fs');
const path = require('path');

const autonomousContent = fs.readFileSync(
  path.join(__dirname, '../../workflows/autonomous.md'), 'utf8');
const auditContent = fs.readFileSync(
  path.join(__dirname, '../../workflows/audit.md'), 'utf8');
const rollbackContent = fs.readFileSync(
  path.join(__dirname, '../../workflows/rollback.md'), 'utf8');

describe('ENH-050: Git tag format includes branch + version', () => {
  describe('workflows/autonomous.md', () => {
    it('defines BRANCH_SAFE resolution', () => {
      expect(autonomousContent).toContain('BRANCH_SAFE');
    });
    it('defines VERSION resolution', () => {
      expect(autonomousContent).toMatch(/VERSION=.*package\.json/);
    });
    it('defines TAG_PREFIX with all three components', () => {
      expect(autonomousContent).toContain('TAG_PREFIX');
      expect(autonomousContent).toContain('BRANCH_SAFE');
    });
    it('task start tag uses TAG_PREFIX pattern', () => {
      expect(autonomousContent).toMatch(/TAG_PREFIX.*vp-p.*t.*task/);
    });
    it('phase complete tag uses TAG_PREFIX pattern', () => {
      expect(autonomousContent).toMatch(/TAG_PREFIX.*vp-p.*complete/);
    });
  });

  describe('workflows/audit.md — regex matches both old and new format', () => {
    it('COMPLETE_TAGS regex matches legacy format (no prefix)', () => {
      const line = auditContent.match(/COMPLETE_TAGS=.*\n/)?.[0] || '';
      const regex = /\^vp-p\[0-9\]/;
      expect(regex.test(line)).toBe(true);
    });
    it('COMPLETE_TAGS regex matches new enriched format (version string)', () => {
      const line = auditContent.match(/COMPLETE_TAGS=[\s\S]*?(?=\n\n)/)?.[0] || auditContent;
      // New format includes \d+\.\d+\.\d+ or 0-9\. for version
      expect(line).toMatch(/0-9\]\+\\.\[0-9\]\+\\.\[0-9\]\+|\\d\+\\\.\d|\[0-9\._-\]/);
    });
    it('PREV_TAG grep includes dot in character class for version strings', () => {
      expect(auditContent).toMatch(/\[a-z0-9\._-\]\+.*vp-p/);
    });
  });

  describe('workflows/rollback.md', () => {
    it('tag grep includes dot in character class', () => {
      expect(rollbackContent).toMatch(/\[a-z0-9\._-\]\+-vp-p/);
    });
  });
});
```
