**C) Guard file fast-path behavior (functional)**
```js
const { execSync } = require('child_process');
const os = require('os');
const path = require('path');
const today = new Date().toISOString().split('T')[0];
const guardFile = path.join(os.tmpdir(), `vp-update-check-${today}.done`);

test('second call uses guard file and exits fast', () => {
  // pre-seed the guard file as "no update"
  fs.writeFileSync(guardFile, JSON.stringify({ updateAvailable: false, latest: '99.0.0', installed: '3.12.1', checkedAt: new Date().toISOString() }));
  
  const start = Date.now();
  try {
    execSync('node bin/vp-tools.cjs check-update --silent', { encoding: 'utf8', stdio: 'pipe' });
    const ms = Date.now() - start;
    expect(ms).toBeLessThan(500); // fast-path < 500ms
  } catch (e) {
    // exit 0 doesn't throw, exit 1 does — either is valid
    const ms = Date.now() - start;
    expect(ms).toBeLessThan(500);
  }
});

test('guard file with updateAvailable=true causes exit 1 and prints latest', () => {
  fs.writeFileSync(guardFile, JSON.stringify({ updateAvailable: true, latest: '99.0.0', installed: '3.12.1', checkedAt: new Date().toISOString() }));
  try {
    execSync('node bin/vp-tools.cjs check-update --silent', { encoding: 'utf8' });
    // should not reach here (exit 1 throws)
    expect(true).toBe(false);
  } catch (e) {
    expect(e.stdout || '').toContain('99.0.0');
    expect(e.status).toBe(1);
  }
});

afterAll(() => {
  // clean up seeded guard file
  try { fs.unlinkSync(guardFile); } catch (_) {}
});
```
