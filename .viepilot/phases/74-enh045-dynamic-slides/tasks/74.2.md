## Implementation Notes

```js
const DESIGN_CONFIGS = {
  'modern-tech': {
    colorPalette: 'navy-electric',   // #1a1f36 bg / #4f6ef7 accent
    layoutStyle:  'modern-tech',
    fontPair:     'Calibri / Calibri Light',
  },
  'enterprise': {
    colorPalette: 'navy-gold',       // #1a1f36 bg / #c9a84c accent
    layoutStyle:  'enterprise',
    fontPair:     'Georgia / Calibri',
  },
  'creative': {
    colorPalette: 'dark-vibrant',    // #0d1020 bg / #f7376e accent
    layoutStyle:  'creative',
    fontPair:     'Calibri / Calibri',
  },
};

/**
 * Choose design config based on project context signals.
 * @param {Object} ctx - { sector?, audience?, tone?, proposalType? }
 * @returns {{ colorPalette, layoutStyle, fontPair }}
 */
function getDesignConfig(ctx = {}) {
  const { sector = '', audience = '', tone = '' } = ctx;
  const lower = `${sector} ${audience} ${tone}`.toLowerCase();
  if (/bank|finance|enterprise|corp|legal|compliance|government/.test(lower))
    return DESIGN_CONFIGS.enterprise;
  if (/startup|creative|design|game|media|agency|art/.test(lower))
    return DESIGN_CONFIGS.creative;
  return DESIGN_CONFIGS['modern-tech']; // default
}
```

Export `getDesignConfig` and `DESIGN_CONFIGS` alongside existing exports.
