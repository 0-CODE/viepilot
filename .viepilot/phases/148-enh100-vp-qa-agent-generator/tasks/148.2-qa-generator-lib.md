# Task 148.2 — lib/qa-generator.cjs

## Objective
Create `lib/qa-generator.cjs` — the adapter-aware generation engine for vp-qa.
This module: detects active adapter, loads the appropriate templates from
`agents/qa-templates/`, substitutes stack-specific rules, and writes agent files
to the correct location in the target project.

## Paths
- `lib/qa-generator.cjs`

## File-Level Plan

### lib/qa-generator.cjs

```js
'use strict';
const fs   = require('fs');
const path = require('path');

/**
 * Read a template file from agents/qa-templates/<adapter>/<name>.md.tmpl
 * Returns '' if not found.
 */
function readTemplate(adapterDir, name) {
  const p = path.join(__dirname, '..', 'agents', 'qa-templates', adapterDir, `${name}.md.tmpl`);
  try { return fs.readFileSync(p, 'utf8'); } catch { return ''; }
}

/**
 * Read a stack rule block from agents/qa-templates/rules/<stack>.md
 * Returns '' if not found (non-fatal — just reduces specificity).
 */
function readStackRules(stack) {
  const p = path.join(__dirname, '..', 'agents', 'qa-templates', 'rules', `${stack}.md`);
  try { return fs.readFileSync(p, 'utf8'); } catch { return ''; }
}

/**
 * Substitute template variables.
 * Variables: {{PROJECT_NAME}}, {{STACK}}, {{STACK_RULES}}, {{BACKEND_DIRS}}, {{PROJECT_ROOT}}
 */
function render(template, vars) {
  return Object.entries(vars).reduce(
    (s, [k, v]) => s.replaceAll(`{{${k}}}`, v),
    template
  );
}

/**
 * Determine where to write agents for a given adapter.
 * Returns { outputDir, isSingleFile, singleFileName? }
 */
function getOutputSpec(adapterId, projectRoot) {
  switch (adapterId) {
    case 'claude-code':
      return { outputDir: path.join(projectRoot, '.claude', 'agents'), isSingleFile: false };
    case 'cursor-agent':
      return { outputDir: path.join(projectRoot, '.cursor', 'rules'), isSingleFile: true, singleFileName: 'qa-checklist.mdc' };
    case 'codex':
      return { outputDir: projectRoot, isSingleFile: true, singleFileName: 'AGENTS.md', appendMode: true };
    case 'antigravity':
      return { outputDir: path.join(projectRoot, '.agents', 'skills'), isSingleFile: false };
    case 'copilot':
      return { outputDir: path.join(projectRoot, '.github', 'agents'), isSingleFile: false };
    default:
      return { outputDir: path.join(projectRoot, '.claude', 'agents'), isSingleFile: false };
  }
}

/**
 * Main generation entry point.
 *
 * @param {object} researchSummary - From vp-qa Phase 1 research:
 *   { projectName, stack, stackVersion, backendDirs, projectSize, scanDomains[] }
 * @param {string} adapterId - 'claude-code' | 'cursor-agent' | 'codex' | 'antigravity' | 'copilot'
 * @param {string} projectRoot - Absolute path to target project root
 * @returns {{ path: string, content: string }[]} - Files to write
 */
function generateQaAgents(researchSummary, adapterId, projectRoot) {
  const { projectName, stack, backendDirs = [], scanDomains = [] } = researchSummary;
  const stackRules = readStackRules(stack);
  const templateVars = {
    PROJECT_NAME:  projectName || 'Project',
    STACK:         stack || 'unknown',
    STACK_RULES:   stackRules,
    BACKEND_DIRS:  backendDirs.join(', ') || 'src/, lib/',
    PROJECT_ROOT:  projectRoot,
    SCAN_DOMAINS:  scanDomains.join(', '),
  };

  const adapterDir = adapterId === 'cursor-agent' ? 'cursor' : adapterId;
  const outputSpec = getOutputSpec(adapterId, projectRoot);
  const results = [];

  if (adapterId === 'claude-code') {
    // Multi-file: orchestrator + one file per scan domain
    const orchestrator = readTemplate('claude-code', 'orchestrator');
    if (orchestrator) {
      results.push({
        path: path.join(outputSpec.outputDir, 'qa-orchestrator.md'),
        content: render(orchestrator, { ...templateVars, SUBAGENT_NAMES: scanDomains.map(d => `qa-${d}-scanner`).join(', ') }),
      });
    }
    for (const domain of scanDomains) {
      const tmpl = readTemplate('claude-code', `${domain}-scanner`);
      if (tmpl) {
        results.push({
          path: path.join(outputSpec.outputDir, `qa-${domain}-scanner.md`),
          content: render(tmpl, templateVars),
        });
      }
    }
  } else {
    // Single-file adapters: combine all domains into one prompt
    const tmpl = readTemplate(adapterDir, 'qa-agent') || readTemplate(adapterDir, 'qa-checklist');
    if (tmpl) {
      const combined = render(tmpl, { ...templateVars, DOMAINS_BLOCK: scanDomains.join('\n- ') });
      const fileName = outputSpec.singleFileName || 'qa-orchestrator.md';
      results.push({
        path: path.join(outputSpec.outputDir, fileName),
        content: combined,
      });
    }
  }

  return results;
}

/**
 * Write generated files to disk (creates parent dirs).
 */
function writeAgentFiles(files) {
  for (const { path: p, content, appendMode } of files) {
    fs.mkdirSync(path.dirname(p), { recursive: true });
    if (appendMode && fs.existsSync(p)) {
      // Codex AGENTS.md: append QA section, avoid duplicating
      const existing = fs.readFileSync(p, 'utf8');
      if (!existing.includes('## QA Agent Instructions')) {
        fs.appendFileSync(p, '\n\n' + content, 'utf8');
      }
    } else {
      fs.writeFileSync(p, content, 'utf8');
    }
  }
}

module.exports = { generateQaAgents, writeAgentFiles, readStackRules, render };
```

**Notes:**
- `researchSummary` is produced by the vp-qa skill's Phase 1 research (run inline in the
  skill's LLM context, then passed to this module via `vp-tools qa-generate` subcommand)
- `scanDomains` array members map directly to template file names
  (e.g. `['completeness', 'security', 'performance']` → looks for `completeness-scanner.md.tmpl`)
- Missing templates are silently skipped (non-fatal)
