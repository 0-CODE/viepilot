describe('Phase 148 — ENH-100: vp-qa agent team generator', () => {

      expect(skill).toMatch(/qa-generator/);

      expect(skill).toMatch(/--run/);
      expect(skill).toMatch(/--target/);

      const results = gen.generateQaAgents(
        { projectName: 'TestApp', stack: 'python', backendDirs: ['app'], scanDomains: ['security'] },
        'codex',

      expect(results.some(r => r.path.includes('AGENTS.md'))).toBe(true);

  agent prompts for stack-aware pattern detection
