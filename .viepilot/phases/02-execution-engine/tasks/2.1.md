## Recovery Logic Pseudocode
```
compliance_preflight(task):
  for path in task.write_scope:
    if matches_compliance_domain(path):
      log("[COMPLIANCE] L3 blocked for this task")
      if not task.recovery_overrides.L3.block:
        WARN: "Compliance path detected but L3.block not set"

execute_with_recovery(task, budget):
  attempt execution
  if PASS: return PASS
  
  # L1 recovery
  while l1_used < budget.l1_max and not recovery_overrides.L1.block:
    try lint_fix()
    l1_used++; log l1_recovery event
    if re_verify() == PASS: return PASS
  
  # L2 recovery
  while l2_used < budget.l2_max and not recovery_overrides.L2.block:
    try targeted_test_fix()
    l2_used++; log l2_recovery event
    if re_verify() == PASS: return PASS
  
  # L3 recovery
  if l3_budget > 0 and not recovery_overrides.L3.block:
    try scope_reduce()
    l3_used++; log l3_recovery event
    if re_verify() == PASS: return PARTIAL_PASS
  
  # Exhausted → control point
  control_point("recovery budget exhausted")
```
