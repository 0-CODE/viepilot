### 3. Trong `applyInstallPlan` (switch cases ~dòng 392-487), thêm case:
```js
case 'rewrite_paths_in_dir': {
  if (dryRun) {
    logs.push(`[dry-run] rewrite_paths_in_dir ${step.dir}: "${step.from}" → "${step.to}" in ${step.glob}`);
    break;
  }
  // Walk dir recursively, rewrite matching files
  const rewriteDir = (dir) => {
    if (!fs.existsSync(dir)) return;
    for (const ent of fs.readdirSync(dir, { withFileTypes: true })) {
      const fullPath = path.join(dir, ent.name);
      if (ent.isDirectory()) {
        rewriteDir(fullPath);
      } else if (ent.isFile() && ent.name.endsWith('.md')) {
        const content = fs.readFileSync(fullPath, 'utf8');
        if (content.includes(step.from)) {
          fs.writeFileSync(fullPath, content.split(step.from).join(step.to), 'utf8');
          logs.push(`rewrite: ${fullPath}`);
        }
      }
    }
  };
  rewriteDir(step.dir);
  break;
}
```

**Note:** Dùng `.split(from).join(to)` thay vì regex để tránh escape issues với path strings.
